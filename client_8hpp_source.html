50  JsonRpcResponse call_method(const id_type &id, const std::string &name, const json &params) {
51  json j = {{"method", name}};
52  if (std::get_if<int>(&id) != nullptr) {
53  j["id"] = std::get<int>(id);
54  } else {
55  j["id"] = std::get<std::string>(id);
56  }
57  if (v == version::v2) {
58  j["jsonrpc"] = "2.0";
59  }
60  if (!params.empty() && !params.is_null()) {
61  j["params"] = params;
62  } else if (v == version::v1) {
63  j["params"] = nullptr;
64  }
65  try {
66  json response = json::parse(connector.Send(j.dump()));
67  if (has_key_type(response, "error", json::value_t::object)) {
68  throw JsonRpcException::fromJson(response["error"]);
69  }
70  if (has_key(response, "result") && has_key(response, "id")) {
71  if (response["id"].type() == json::value_t::string)
72  return JsonRpcResponse{response["id"].get<std::string>(), response["result"].get<json>()};
73  else
74  return JsonRpcResponse{response["id"].get<int>(), response["result"].get<json>()};
75  }
76  throw JsonRpcException(-32603, R"(invalid server response: neither "result" nor "error" fields found)");
77  } catch (json::parse_error &e) {
78  throw JsonRpcException(-32700, std::string("invalid JSON response from server: ") + e.what());

49  if (!params.empty() && !params.is_null()) {

52  j["params"] = nullptr;
53  }
54  try {
55  json response = json::parse(connector.Send(j.dump()));
56  if (has_key_type(response, "error", json::value_t::object)) {

58  }
59  if (has_key(response, "result") && has_key(response, "id")) {
